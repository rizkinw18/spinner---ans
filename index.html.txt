<script>
let angle = 0;
const hadiah = [
  "🎁 Treatment Kit",
  "🛞 Semir Ban",
  "🎒 Merchpack ANS"
];

function spin() {
  const spinAngle = Math.floor(Math.random() * 360) + 1080;
  angle += spinAngle;

  document.getElementById("wheel").style.transform =
    `rotate(${angle}deg)`;

  setTimeout(() => {
    const actualAngle = angle % 360;
    let index;

    if (actualAngle < 120) index = 0;
    else if (actualAngle < 240) index = 1;
    else index = 2;

    document.getElementById("result").innerText =
      "Kamu mendapatkan: " + hadiah[index];
  }, 4000);
}
</script>
